816 double meshhelp::ProjectRay(int count, const W &&boundBox, const T &dir, const V &orig, double minDist)
817 {
818  /*
819  Calculates the distance to project a single ray to reach the bounding box
820 
821  It is templated to accept any types of containers with `count` elements,
822  accessible by operator[];
823 
824  W is an object of size [2][count] accessible by operators [][]
825  */
826 
827  double l = -INFINITY;
828 
829  for (int i = 0; i < count; ++i)
830  {
831  l = std::max(l, std::min((boundBox[0][i] - orig[i]) / dir[i], (boundBox[1][i] - orig[i]) / dir[i]));
832  }
833  return std::min(l, minDist);
834 }
